	QUESTION ON SELECT
1757. Recyclable and Low Fat Products
ANS: SELECT PRODUCT_ID FROM PRODUCTS WHERE LOW_FATS = 'Y' AND RECYCLABLE = 'Y'; 
584. Find Customer Referee
ANS: SELECT NAME FROM CUSTOMER WHERE REFEREE_ID <> 2 OR REFEREE_ID IS NULL;
595. Big Countries
ANS : SELECT NAME, POPULATION, AREA FROM WORLD WHERE AREA >= 3000000 OR POPULATION >= 25000000;
1148. Article Views I
ANS:SELECT DISTINCT(AUTHOR_ID) ID FROM VIEWS WHERE AUTHOR_ID = VIEWER_ID ORDER BY AUTHOR_ID ASC;
